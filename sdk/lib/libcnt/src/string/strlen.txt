
#include <string.h>

#ifdef _MSC_VER
#pragma function(strlen)
#elif defined(__GNUC__)
#pragma GCC optimize("no-tree-loop-distribute-patterns")
#endif

#if defined(__clang__)
__attribute__((no_builtin("strlen")))
#endif
size_t
strlen1(
    const char* _Str)
{
    const char* p = _Str;

    while (*p)
        p++;

    return (p - _Str);
}

#if defined(__clang__)
__attribute__((no_builtin("strlen")))
#endif
size_t
strlen2(
    const char* _Str)
{
    size_t count = 0;

    while (_Str[count])
        count++;

    return count;
}

#if defined(__clang__)
__attribute__((no_builtin("strlen")))
#endif
size_t
strlen3(
    const char* _Str)
{
    size_t count = -1;

    do
    {
       count++;
    } while (_Str[count]);
 
    return count;
}
